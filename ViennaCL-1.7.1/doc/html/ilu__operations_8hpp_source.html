219 template<typename NumericT>
220 void ilu_transpose(compressed_matrix<NumericT> const & A,
221  compressed_matrix<NumericT> & B)
222 {
223  viennacl::context orig_ctx = viennacl::traits::context(A);
224  viennacl::context cpu_ctx(viennacl::MAIN_MEMORY);
225  (void)orig_ctx;
226  (void)cpu_ctx;
227 
228  viennacl::compressed_matrix<NumericT> A_host(0, 0, 0, cpu_ctx);
229  (void)A_host;
230 
231  switch (viennacl::traits::handle(A).get_active_handle_id())
232  {
233  case viennacl::MAIN_MEMORY:
234  viennacl::linalg::host_based::ilu_transpose(A, B);
235  break;
236 #ifdef VIENNACL_WITH_OPENCL
237  case viennacl::OPENCL_MEMORY:
238  A_host = A;
239  B.switch_memory_context(cpu_ctx);
240  viennacl::linalg::host_based::ilu_transpose(A_host, B);
241  B.switch_memory_context(orig_ctx);
242  break;
243 #endif
244 #ifdef VIENNACL_WITH_CUDA
245  case viennacl::CUDA_MEMORY:
246  A_host = A;
247  B.switch_memory_context(cpu_ctx);
248  viennacl::linalg::host_based::ilu_transpose(A_host, B);
249  B.switch_memory_context(orig_ctx);
250  break;
251 #endif
252  case viennacl::MEMORY_NOT_INITIALIZED:
253  throw memory_exception("not initialised!");
254  default:
255  throw memory_exception("not implemented");
256  }
