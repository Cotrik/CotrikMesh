141 template <typename T>
142 void cdiv(T xr, T xi, T yr, T yi, T& cdivr, T& cdivi)
143 {
144  // Complex scalar division.
145  T r;
146  T d;
147  if (std::fabs(yr) > std::fabs(yi))
148  {
149  r = yi / yr;
150  d = yr + r * yi;
151  cdivr = (xr + r * xi) / d;
152  cdivi = (xi - r * xr) / d;
153  }
154  else
155  {
156  r = yr / yi;
157  d = yi + r * yr;
158  cdivr = (r * xr + xi) / d;
159  cdivi = (r * xi - xr) / d;
160  }
161 }
