71  inline vcl_size_t next_power_2(vcl_size_t n)
72  {
73  n = n - 1;
74 
75  vcl_size_t power = 1;
76 
77  while (power < sizeof(vcl_size_t) * 8)
78  {
79  n = n | (n >> power);
80  power *= 2;
81  }
82 
83  return n + 1;
84  }
