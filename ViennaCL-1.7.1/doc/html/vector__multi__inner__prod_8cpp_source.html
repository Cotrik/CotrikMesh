78 template<typename ScalarType, typename ViennaCLVectorType>
79 ScalarType diff(std::vector<ScalarType> const & v1, ViennaCLVectorType const & vcl_vec)
80 {
81  std::vector<ScalarType> v2_cpu(vcl_vec.size());
82  viennacl::backend::finish();
83  viennacl::copy(vcl_vec, v2_cpu);
84 
85  for (unsigned int i=0;i<v1.size(); ++i)
86  {
87  if ( std::max( std::fabs(v2_cpu[i]), std::fabs(v1[i]) ) > 0 )
88  v2_cpu[i] = std::fabs(v2_cpu[i] - v1[i]) / std::max( std::fabs(v2_cpu[i]), std::fabs(v1[i]) );
89  else
90  v2_cpu[i] = 0.0;
91  }
92 
93  ScalarType norm_inf = 0;
94  for (std::size_t i=0; i<v2_cpu.size(); ++i)
95  norm_inf = std::max<ScalarType>(norm_inf, std::fabs(v2_cpu[i]));
96 
97  return norm_inf;
98 }
