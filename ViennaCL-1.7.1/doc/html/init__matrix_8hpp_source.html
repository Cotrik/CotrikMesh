23 static ViennaCLStatus init_cuda_matrix(viennacl::backend::mem_handle & h, ViennaCLMatrix A)
24 {
25 #ifdef VIENNACL_WITH_CUDA
26  h.switch_active_handle_id(viennacl::CUDA_MEMORY);
27  h.cuda_handle().reset(A->cuda_mem);
28  h.cuda_handle().inc();
29  if (A->precision == ViennaCLFloat)
30  h.raw_size(static_cast<viennacl::vcl_size_t>(A->internal_size1) * static_cast<viennacl::vcl_size_t>(A->internal_size2) * sizeof(float)); // not necessary, but still set for conciseness
31  else if (A->precision == ViennaCLDouble)
32  h.raw_size(static_cast<viennacl::vcl_size_t>(A->internal_size1) * static_cast<viennacl::vcl_size_t>(A->internal_size2) * sizeof(double)); // not necessary, but still set for conciseness
33  else
34  return ViennaCLGenericFailure;
35 
36  return ViennaCLSuccess;
37 #else
38  (void)h;
39  (void)A;
40  return ViennaCLGenericFailure;
41 #endif
42 }
